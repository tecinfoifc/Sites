// Solução para o exercício 2 da Prova de Aquecimento da OBI2015
// Economia Brasileira

var i,     // contador
    q,     // número de eleitores
    v,     // variável para ler um voto
    votos; // número de votos a favor

// inicializa 
votos = 0;

// lê a entrada e contabiliza votos a favor
scanf("%d", "q");
for(i=0; i<q; i++) {
    scanf("%d", "v");
    if (v===0)
	votos++;
}

// escreve o resultado
if (votos > q/2)
    printf("S\n");
else
    printf("N\n");
